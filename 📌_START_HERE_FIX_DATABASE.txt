HOW TO FIX (3 SIMPLE STEPS)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 1: GET PLANETSCALE CREDENTIALS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Go to: https://app.planetscale.com
2. Login with your account
3. Click your database: "school-management"
4. Click: "Connect" button
5. Select: "Node.js" from dropdown
6. Copy these values:
   - DB_HOST = aws.connect.psdb.cloud
   - DB_USER = (your username)
   - DB_PASSWORD = (your password)
   - DB_NAME = school-management

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
STEP 2: SET VARIABLES ON RENDER
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Go to: https://dashboard.render.com
2. Click: "school-management-backend-gnav"
3. Click: "Environment" tab
4. Add these variables (click "Add Environment Variable" for each):

   DB_HOST = aws.connect.psdb.cloud
   DB_USER = (your PlanetScale username)
   DB_PASSWORD = (your PlanetScale password)
   DB_NAME = school-management
   DB_PORT = 3306
   NODE_ENV = production
   JWT_SECRET = your_super_secret_jwt_key_123_change_in_production
   FRONTEND_URL = https://school-management-system.vercel.app

5. Click: "Redeploy" button (top right)
6. Wait: 2-3 minutes

STEP 3: VERIFY CONNECTION
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

2. Look for: "Connected to MySQL database"
3. If you see this, it's working! ✅
